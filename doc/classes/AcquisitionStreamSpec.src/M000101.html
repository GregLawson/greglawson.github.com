def uriHash
        componentNames=@uri.class::COMPONENT
        hash={}
        componentNames.each_index do |i| 
                componentName= componentNames[i] 
                component=uriComponent(componentName)
                if !component[0].nil? then
                        hash.merge!({ componentName => component })
                end #if
        end # each_index
        return hash
end
