def find_or_initialize(findCriteria)
        records=find(:all,findCriteria)
        if records.empty? then
                ret= self.new(findCriteria)
                return ret
        elsif records.size==1 then
                return records[0]
        else
                @@log.debug("criteria not unque; records=#{records.inspect}")
                raise 
        end
end
