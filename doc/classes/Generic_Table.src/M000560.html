def requireColumn(name,type)
#       Global::log.info("self.class=#{self.class}")
#       Global::log.info("name=#{name}")
        if has_attribute?(name) then
                return ""
        else
                puts "Column #{name} to be created with #{type}" if $VERBOSE
                return addColumn(name,type)
        end
end
