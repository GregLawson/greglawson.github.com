 def initialize(params=nil)
        @frozen=false
        @thawLength=0
        @freezeLength=0
        @previousThaw=0
        @previousFreeze=0
 
        super(params)  

        if self[:shortest_update].nil? then
                self[:shortest_update]=100*365*24*3600
        end
        if self[:longest_no_update].nil? then
                self[:longest_no_update]=0
        end

end
