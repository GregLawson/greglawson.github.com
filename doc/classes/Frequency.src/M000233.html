def checkFreeze(acquisitionsUpdated=acquisitionsUpdated?)
        if acquisitionsUpdated then
                if @frozen then #begin thaw
                        @frozen=false
                        @previousThaw=@thawLength    
                        @thawLength=1
                else 
                        @thawLength+=1
                end
        else
                if @frozen then
                        @freezeLength+=1
                else
                        @frozen=true
                        @previousFreeze=@freezeLength
                        @freezeLength=1
                end
        end
        return @freezeLength
        
end
